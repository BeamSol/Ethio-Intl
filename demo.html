How to use:

// Import the hook
import { useTransliterate } from 'ethio-intl';

function MyComponent() {
  const [input, setInput] = useState('');
  const amharic = useTransliterate(input);

  return (
    <div>
      <input
        value={input}
        onChange={(e) => setInput(e.target.value)}
        placeholder="Type in English"
      />
      <div>{amharic}</div>
    </div>
  );
}
